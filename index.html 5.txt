// ✨ Open first gift
function openGift(){
  document.getElementById("gift").style.display="none";
  document.getElementById("intro").style.display="none";
  document.getElementById("content").style.display="block";
  typeWriter("message","يا Nada… كل سنة وإنتي طيبة وعيد سعيد ❤️",40,function(){
    document.getElementById("loveQuestion").style.display="block";
  });
}

// 💌 Typing effect
function typeWriter(id,msg,speed,callback){
  let element = document.getElementById(id);
  element.innerHTML = "";
  let index = 0;
  function typing(){
    if(index < msg.length){
      element.innerHTML += msg.charAt(index);
      index++;
      setTimeout(typing,speed);
    } else if(callback){ callback(); }
  }
  typing();
}

// 🪄 Move "No" button
function moveButton(){
  let btn=document.getElementById("no");
  btn.style.top=Math.random()*80+"%";
  btn.style.left=Math.random()*80+"%";
}

// 📸 Show second page
function showSecondPage(){
  document.getElementById("content").style.display="none";
  document.getElementById("secondPage").style.display="block";
}

// ⏳ Show secret days counter
function showSecret(){
  let startDate=new Date("2022-08-20");
  let today=new Date();
  let diff=today-startDate;
  let days=Math.floor(diff/(1000*60*60*24));
  document.getElementById("daysCounter").innerHTML="بقالنا "+days+" يوم مع بعض ❤️";
  createHearts();
}

// 🧡 Floating hearts
function createHearts(){
  for(let i=0;i<60;i++){
    let heart=document.createElement("div");
    heart.className="heart";
    heart.innerHTML="❤️";
    heart.style.left=Math.random()*100+"vw";
    heart.style.fontSize=(12+Math.random()*20)+"px";
    heart.style.animationDuration=(4+Math.random()*6)+"s";
    document.body.appendChild(heart);
  }
}
